var examplearr = [92,1, 4, 55, 32, 21, 44, 2]; function sortarr(arr) { for (var i = 0; i < arr.length - 1; i++) { for (var j = 0; j < arr.length - 1 - i; j++) { if (arr[j] > arr[j + 1]) { var temp = arr[j]; arr[j] = arr[j + 1]; arr[j + 1] = temp; } } } document.write(arr); } sortarr(examplearr);
